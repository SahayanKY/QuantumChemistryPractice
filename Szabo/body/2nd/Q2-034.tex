\subsection{問}
\begin{align}
	[s^2, s_z]
&=
	0
\end{align}
を示せ。

\subsection{解}
\begin{align}
	[s^2 ,s_z]
&=
	s^2 s_z
	-
	s_z s^2 \\
%
%
&=
	s_x^2 s_z
	-
	s_z s_x^2
	+
	s_y^2 s_z
	-
	s_z s_y^2
	+
	s_z^2 s_z
	-
	s_z s_z^2 \\
%
%
&=
	s_x
	(
		s_z s_x
		-
		i s_y
	)
	-
	(
		s_x s_z
		+
		i s_y
	)
	s_x
	+
	s_y
	(
		s_z s_y
		+
		i s_x
	)
	-
	(
		s_y s_z
		-
		i s_x
	)
	s_y \\
%
%
&=
	-
	i s_x s_y
	-
	i s_y s_x
	+
	i s_y s_x
	+
	i s_x s_y \\
%
%
&=
	0
\end{align}
である。
